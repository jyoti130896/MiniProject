TABLE -- COURSE_MASTER

CREATE TABLE COURSE_MASTER 
(
COURSE_ID NUMBER(5) PRIMARY KEY,
COURSE_NAME VARCHAR(50) NOT NULL,
NO_OF_DAYS NUMBER(5) NOT NULL
);


TABLE -- EMPLOYEE_MASTER

CREATE TABLE EMPLOYEE_MASTER
(
EMPLOYEE_ID NUMBER(5) PRIMARY KEY,
EMPLOYEE_NAME VARCHAR(50) NOT NULL,
PASSWORD VARCHAR(20) NOT NULL,
ROLE VARCHAR(20) NOT NULL
);

TABLE -- TRAINING_MASTER

CREATE TABLE TRAINING_MASTER 
(
TRAINING_CODE NUMBER(5) PRIMARY KEY,
COURSE_CODE NUMBER(5) FOREIGN KEY REFERENCES COURSE_MASTER(COURSE_ID),
FACULTY_CODE NUMBER(5) FOREIGN KEY REFERENCES EMPLOYEE_MASTER(EMPLOYEE_ID),
START_DATE DATE NOT NULL,
END_DATE DATE NOT NULL
);


TABLE -- TRAINING_PARTICIPANT_MASTER

CREATE TABLE TRAINING_PARTICIPANT_MASTER
(
TRANING_CODE NUMBER(5) FOREIGN KEY REFERENCES TRAINING_MASTER(TRAINING_CODE),
PARTICIPANT_ID NUMBER(5) FOREIGN KEY REFERENCES EMPLOYEE_MASTER(EMPLOYEE_ID)
);

TABLE -- FACULTY_SKILL

CREATE TABLE FACULTY_SKILL 
(
FACULTY_ID NUMBER(5) FOREIGN KEY REFERENCES EMPLOYEE_MASTER(EMPLOYEE_ID),
SKILL_SET VARCHAR2(200) NOT NULL
);

TABLE -- FEEDBACK_MASTER

CREATE TABLE FEEDBACK_MASTER
(
TRAINING_CODE NUMBER(5) FOREIGN KEY REFERENCES TRAINING_MASTER(TRAINING_CODE),
PARTICIPANT_ID NUMBER(5) FOREIGN KEY REFERENCES EMPLOYEE_MASTER(EMPLOYEE_ID),
FB_PRS_COMM NUMBER(1) NOT NULL,
FB_CLRFY_DBTS NUMBER(1) NOT NULL,
FB_TM NUMBER(1) NOT NULL,
FB_HND_OUT NUMBER(1) NOT NULL,
FB_HW_SW_NTWRK NUMBER(1) NOT NULL,
COMMENTS VARCHAR2(200) NOT NULL,
SUGGESTIONS VARCHAR2(200) NOT NULL
);
